/**function */

export function getUrlExtension(url: string): string {
  return url.split(/[#?]/)[0].split('.').pop().trim()
}

export function getUrlFileName(url: string): string {
  const filenames = url.split('/')
  return filenames[filenames.length - 1].trim()
}
